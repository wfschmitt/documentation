def collect_dir_tree(path)
  @dirs=[]
  path = Pathname.new(path) unless path.is_a? Pathname
  files = find(path)
  files.each do |page|
    @dirs << page if page.directory?
  end
  @dirs
end

def collect_md_tree(path)

def collect_sorted_readme(path)
  @readmes=[]
  path = Pathname.new(path) unless path.is_a? Pathname
  files = find(path)
  files.each do |page|
    new_page = filter(page.to_s, FIND_FILTER_MARKDOWN_README)
    @readmes << new_page unless new_page.nil?
  end
  @readmes = @readmes.sort
  @readmes = @readmes.uniq
  return sorted_pages(@readmes)
end

def filter(page, pattern)
  page = Pathname.new(page)
  return nil if page.directory?
  base = page.basename.to_s.downcase
  pattern.each do |pat|
    return nil unless base.include?(pat)
  end
  # puts "Found : #{page.to_path}".cyan if verbose?
  return page
end

def find(path)
  @@pages=[]
  path = Dir.getwd if path.equal?('.')
  path = Pathname.new(path) unless path.is_a? Pathname
  Dir.chdir(path) do
    rule = File::Find.new
    tree = rule.find
    tree.each do |p|
      @@pages << Pathname.new(p).relative_path_from(path)
    end
    return @@pages
  end
end

def sorted_pages(pages)
  tops1=[]
  tops2=[]
  tops3=[]
  tops4=[]

  pages.each do |p|
    ps = p.to_s
    if ps.downcase.include?('doc/readme.md')
      tops1 << ps
    elsif ps.downcase.include?('generic/doc')
      tops2 << ps
    elsif ps.downcase.include?('generic/')
      tops3 << ps
    else
      tops4 << ps
    end
  end
  plebs = tops2 + tops3 + tops4
  [tops1, plebs]
end
